自己动手实现一个kubectl exec

2020-06-20
7 minute read
k8s

在日常工作中kubectl exec 可以说是非常高频使用的，如果你想自己了解相关原理，不妨自己动手写一个。

知识储备：

  • websocket 阮一峰这篇《WebSocket 教程- 阮一峰的网络日志》写的比较详进。

  • kubectl exec 原理

    https://itnext.io/how-it-works-kubectl-exec-e31325daa910

    https://erkanerol.github.io/post/how-kubectl-exec-works/

    如果你英文阅读能还可以，这两篇文章从原理方面介绍了exec是如何工作的。

了解了以上知识之后，接下来我们就开始动手吧。

首先来初始化一下项目，这里使用go mod作为依赖管理工具。k8s的client-go对机器版本是有要求的，所以在初始化的时候最好去官方那边找一下可用的版本。如果遇到mod/k8s.io/client-go@v10.0.0+incompatible/kubernetes/scheme/register.go:22:2: unknown import path "k8s.io/api/admissionregistration /v1alpha1": cannot find module providing package k8s.io/api/admissionregistration/v1alpha1

这种报错，可以尝试强制指定版本，这个也是从kubebuilder那里学到的。

现在就开始做吧，从上面的URL可以看到exec 是属于pod的资源，

 1// 初始化pod所在的corev1资源组，发送请求
 2// PodExecOptions struct 包括Container stdout stdout  Command 等结构
 3// scheme.ParameterCodec 应该是pod 的GVK （GroupVersion & Kind）之类的
 4req := clientset.CoreV1().RESTClient().Post().
 5		Resource("pods").
 6		Name("nginx-8486565b79-4hb5t").
 7		Namespace("default").
 8		SubResource("exec").
 9		VersionedParams(&corev1.PodExecOptions{
10			Command: []string{"bash"},
11			Stdin:   true,
12			Stdout:  true,
13			Stderr:  true,
14			TTY:     false,
15		}, scheme.ParameterCodec)
16
17// remotecommand 主要实现了http 转 SPDY 添加X-Stream-Protocol-Version相关header 并发送请求
18	exec, err := remotecommand.NewSPDYExecutor(config, "POST", req.URL())
19
20// 建立链接之后从请求的sream中发送、读取数据
21	if err = exec.Stream(remotecommand.StreamOptions{
22		Stdin:  os.Stdin,
23		Stdout: os.Stdout,
24		Stderr: os.Stderr,
25		Tty:    false,
26	}); err != nil {
27		fmt.Print(err)
28	}

完整示例

 1package main
 2
 3import (
 4	"flag"
 5	"fmt"
 6	"io"
 7	"os"
 8	"path/filepath"
 9
10	"golang.org/x/crypto/ssh/terminal"
11	corev1 "k8s.io/api/core/v1"
12	"k8s.io/client-go/kubernetes"
13	"k8s.io/client-go/kubernetes/scheme"
14	"k8s.io/client-go/tools/clientcmd"
15	"k8s.io/client-go/tools/remotecommand"
16	"k8s.io/client-go/util/homedir"
17)
18
19func main() {
20
21	var kubeconfig *string
22	if home := homedir.HomeDir(); home != "" {
23		kubeconfig = flag.String("kubeconfig", filepath.Join(home, ".kube", "vm"), "(optional) absolute path to the kubeconfig file")
24	} else {
25		kubeconfig = flag.String("kubeconfig", "", "absolute path to the kubeconfig file")
26	}
27	flag.Parse()
28
29	config, err := clientcmd.BuildConfigFromFlags("", *kubeconfig)
30	if err != nil {
31		panic(err)
32	}
33	clientset, err := kubernetes.NewForConfig(config)
34	if err != nil {
35		panic(err)
36	}
37
38	// 初始化pod所在的corev1资源组，发送请求
39	// PodExecOptions struct 包括Container stdout stdout  Command 等结构
40	// scheme.ParameterCodec 应该是pod 的GVK （GroupVersion & Kind）之类的
41	req := clientset.CoreV1().RESTClient().Post().
42		Resource("pods").
43		Name("nginx-8486565b79-4hb5t").
44		Namespace("default").
45		SubResource("exec").
46		VersionedParams(&corev1.PodExecOptions{
47			Command: []string{"bash"},
48			Stdin:   true,
49			Stdout:  true,
50			Stderr:  true,
51			TTY:     false,
52		}, scheme.ParameterCodec)
53
54	// remotecommand 主要实现了http 转 SPDY 添加X-Stream-Protocol-Version相关header 并发送请求
55	exec, err := remotecommand.NewSPDYExecutor(config, "POST", req.URL())
56
57	// 检查是不是终端
58	if !terminal.IsTerminal(0) || !terminal.IsTerminal(1) {
59		 fmt.Errorf("stdin/stdout should be terminal")
60	}
61	// 这个应该是处理Ctrl + C 这种特殊键位
62	oldState, err := terminal.MakeRaw(0)
63	if err != nil {
64		 fmt.Println(err)
65	}
66	defer terminal.Restore(0, oldState)
67
68	// 用IO读写替换 os stdout
69	screen := struct {
70		io.Reader
71		io.Writer
72	}{os.Stdin, os.Stdout}
73
74	// 建立链接之后从请求的sream中发送、读取数据
75	if err = exec.Stream(remotecommand.StreamOptions{
76		Stdin:  screen,
77		Stdout: screen,
78		Stderr: screen,
79		Tty:    false,
80	}); err != nil {
81		fmt.Print(err)
82	}
83}
