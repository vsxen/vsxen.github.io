11# 为Pod template 加上新label 打patch
12kubectl patch deployment simple --type=json -p='[{"op": "add", "path": "/spec/template/metadata/labels/this", "value": "that"}]'

32# 官方的示例

34Examples:
35  # Partially update a node using a strategic merge patch. Specify the patch as JSON.
36  kubectl patch node k8s-node-1 -p '{"spec":{"unschedulable":true}}'
37
38  # Partially update a node using a strategic merge patch. Specify the patch as YAML.
39  kubectl patch node k8s-node-1 -p $'spec:\n unschedulable: true'
40
41  # Partially update a node identified by the type and name specified in "node.json" using strategic merge patch.
42  kubectl patch -f node.json -p '{"spec":{"unschedulable":true}}'
43
44  # Update a container's image; spec.containers[*].name is required because it's a merge key.
45  kubectl patch pod valid-pod -p '{"spec":{"containers":[{"name":"kubernetes-serve-hostname","image":"new image"}]}}'

 4# 删除非running Pod
 5kubectl get pods --all-namespaces -ojson | jq -r '.items[] | select(.status.reason!=null) | select(.status.reason | contains("Evicted")) | .metadata.name + " " + .metadata.namespace' | xargs -n2 -l bash -c 'kubectl delete pods $0 --namespace=$1'
